2. This program prints pounds symbols in the shape of a right triangle and the user input dictates the height of the triangle (i.e. how many rows of pound symbols the triangle will have). 


3. 


#include <iostream>
using namespace std;


int main()
{
        int side;


        cout << "Enter a number: ";
        cin >> side;
        int counter; 


        for (int i = 0; i < side; i++)
        {
                counter = i;                       //counter is used to keep track of how many # per row 
                while (counter >= 0)           //while counter is greater than or equal to 0 print #
                {
                        cout << "#"; 
                        counter--; 
                }
                cout << "\n";
        }
}
